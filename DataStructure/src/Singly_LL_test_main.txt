#include <cstdlib>
#include <iostream>
#include "SinglyLinkedList.h"
using namespace std;

/*
 * 
 */
int main(int argc, char** argv) {
    
    SinglyLinkedList* LL = new SinglyLinkedList();
    LL->insertAtFirst(8);
    LL->print_LL();
    LL->insertAtFirst(3);
    LL->insertAtTail(34);
    LL->print_LL();
    LL->insertAtPos(3, 45);
    LL->print_LL();
    LL->remove(45);
    LL->print_LL();

    return 0;
}
